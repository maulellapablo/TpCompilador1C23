*------------ AND  -----------*

a = 1
b = 1
c = 2

if (a > b & c > b)
{
    write("a es mas grande que b y c es mas grande que b")
}

*------------ OR  -----------*

if (a > b || c > b)
{
    write("a es mas grande que b o c es mas grande que b")
}

*------------ NOT  -----------*

if (not a > b)
{
    write("a no es mas grande que b")
}
